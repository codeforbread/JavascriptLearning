//To check if person can vote in election based on age
//Condition :- Any person with age higher than or equal to 18 years can vote in election. 
age = 20;

///if(condition) compare age with 18 if less it returns true otherwise returns false
if(age < 18){
    document.write("You cannot vote for election");
} else {
    document.write("You can vote for election");
}
//output
You can vote for election

//Loops are used to repeat the task till condition satisfies.
//print numbers from 1 to 10

//if we remove number = number + 1; while loop will run infinetly
